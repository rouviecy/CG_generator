cmake_minimum_required(VERSION 2.8)
project(cg_generator)

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")

add_library(CG_GENERATOR
    CPP/readonly/common_readonly.h
    CPP/readonly/Game_readonly.h
    CPP/readonly/Game_readonly.cpp
    CPP/readonly/Entity_readonly.h
    CPP/readonly/Entity_readonly.cpp
    CPP/readonly/entities/Ship_readonly.h
    CPP/readonly/entities/Ship_readonly.cpp
    CPP/readonly/entities/Barrel_readonly.h
    CPP/readonly/entities/Barrel_readonly.cpp
    CPP/readonly/entities/Ball_readonly.h
    CPP/readonly/entities/Ball_readonly.cpp
    CPP/readonly/entities/Mine_readonly.h
    CPP/readonly/entities/Mine_readonly.cpp
    CPP/custom/common.h
    CPP/custom/Game.h
    CPP/custom/Game.cpp
    CPP/custom/Entity.h
    CPP/custom/Entity.cpp
    CPP/custom/entities/Ship.h
    CPP/custom/entities/Ship.cpp
    CPP/custom/entities/Barrel.h
    CPP/custom/entities/Barrel.cpp
    CPP/custom/entities/Ball.h
    CPP/custom/entities/Ball.cpp
    CPP/custom/entities/Mine.h
    CPP/custom/entities/Mine.cpp
)

add_executable(cg CPP/readonly/main.cpp)
target_link_libraries(cg CG_GENERATOR)

add_executable(cgcode CPP/readonly/mainMono.cpp)
set_target_properties(cgcode PROPERTIES COMPILE_FLAGS "-E -P -DEXPLICITE_EXCLUDE")
